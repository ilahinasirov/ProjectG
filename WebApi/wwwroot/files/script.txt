USE [ProjectG]
GO

DECLARE @i INT = 1;
DECLARE @departmentId INT;
DECLARE @time DATETIME2 = SYSDATETIME();

WHILE @i <= 90
BEGIN
    SET @departmentId = ((@i - 1) % 3) + 1;

    INSERT INTO [dbo].[Requests] (BaseId, DepartmentId, ConfirmationCount, Time)
    VALUES (@i, @departmentId, 0, @time);

    SET @i = @i + 1;
END
GO
